project(clmtl C CXX)

add_subdirectory(deps)

add_library(clmtl
        src/Driver.cpp
        src/Dispatch.h
        src/Dispatch.cpp
        src/Util.h
        src/Util.cpp
        src/Platform.h
        src/Platform.cpp
        src/LibraryPool.h
        src/LibraryPool.cpp
        src/Device.h
        src/Device.cpp
        src/Object.h
        src/Object.cpp
        src/Context.h
        src/Context.cpp
        src/CommandQueue.h
        src/CommandQueue.cpp
        src/Buffer.h
        src/Buffer.cpp
        src/Program.h
        src/Program.cpp
        src/Reflector.h
        src/Reflector.cpp
        src/Kernel.h
        src/Kernel.cpp
        src/Sampler.h
        src/Sampler.cpp
)

target_compile_features(clmtl
    PUBLIC
        cxx_std_20
)

target_compile_definitions(clmtl
    PUBLIC
        CL_TARGET_OPENCL_VERSION=100
)

target_link_libraries(clmtl
    PUBLIC
        khronos
        apple
        clspv
        json
)

add_subdirectory(demo)
